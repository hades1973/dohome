\begin{frame}[plain]
	(1) 计算$M_2$
	\uncover<2->{
\vspace{-0.8em} {\small 
\begin{align*}
	a_s &= {{.Q.D_s}}mm \\
	h_0 &= h- a_s = {{.Q.H}} - {{.Q.D_s}} = {{.H_0}}mm \\
	M_2 &= M - f'_y A'_{s} (h_0 - a'_s)  \\
            &= {{.Q.M}}\times 10^6 - {{.Fy}}\times{{.Q.Ass}}\times({{.H_0}} - {{.Q.D_ss}}) N\cdot m\\
            &={{printf "%4.1f" .M_2}}kN\cdot m \\
%%%			
%%%{{if .M_2Negative}} 
	M_2 &< 0, \text{受压纵筋不屈服，取}x=2a'_s, \\ 
	    &\text{使用对受压钢筋合力点取矩的方程，有： }\\ 
	A_s &= \frac{M}{f_y (h_0 - a'_s)} \\
            &= \frac{ {{.Q.M}}\times10^6 }{ {{.Fy}}\times( {{.H_0}} - {{.Q.D_ss}} ) } \\
            &= {{printf "%4.1f" .As}}mm^2 
\end{align*}

\begin{align*}
	\alpha_s &= \frac{M_2}{\alpha_1 fc b h^2_0} 
	= \frac{ {{printf "%4.1f" .M_2}}\times10^6 }{1.0\times {{.Fc}}\times {{.Q.B}}\times {{.H_0}}^2}= {{printf "%4.3f" .Alpha_s}} \\
	\alpha_{s,max} &= \xi_b(1-0.5\xi_b) = {{printf "%4.3f" .Alpha_smax}} \\

	\xi &= 1- \sqrt{1 - 2\alpha_s} = 1 - \sqrt{1- 2\times {{printf "%4.3f" .Alpha_s}} } = {{printf "%4.3f" .Xi}} \\
